select d.dep_id,d.dept_name,e.emp_sal as salary,
case when d.dept_id not in e.dept_id then e.emp_sal=0
end as salary
from department d join employee e on d.dept_id=e.dept_id group by dept_name;

2.
select d.dept_id,d.dept_name,sum(emp_sal) as sum_emp_sal,
case when e.dept_id=3 then 'Others'
end 
from employee e join department d group by d.dept_id;
